def void change_int(var int a) {
    a = (3 + 11) / 2 + 4;
}

def bool is_even(int a) {
    if (a % 2 == 0) {
        return true;
    }
    else {
        return false;
    }
}

def int main() {
    var int a = 10;
    printBool(is_even(a));
    var bool b = false;

    change_int(b);
    printInt(a);
    printBool(is_even(b));
    return 0;
}
